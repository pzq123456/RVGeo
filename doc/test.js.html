/**
 * 生成圆形点集
 * @param {number} number - 点集大小
 * @param {number} radius - 圆形区域的半径
 * @returns {array} [[x1,y1],[x2,y2]...[xn,yn]]
 */
export function test_6(number,radius){
    let vertices = new Array(number)
    var x,y,i;
    for( i = vertices.length; i--; ) {
        do {
             x = Math.random() - 0.5;
              y = Math.random() - 0.5;
        } while(x * x + y * y > 0.25);

        x = (x * 0.96875 + 0.5) * radius;
        y = (y * 0.96875 + 0.5) * radius;

        vertices[i] = [x, y];
    }
    return vertices;
}

/**
 * 在一组之中随机选出n个值
 * @param {array} arr 
 * @param {number} n 
 */
export function test_8(arr,n){
    if(n > arr.length ){
        return [];
    }
    else if(n === arr.length){
        let num = arr.length - 1;
        let arr1 = arr.slice();
        let res = [];

        for(let i = 0 ;i< n - 1;i++){
            let ren = kernel_8(num);
            let itm = arr1[ren];
            res.push(itm);
            arr1.splice(ren,1);
            num = num -1;
        }
        res.push(arr1[0]);
        return res;
    }
    else{
        let num = arr.length - 1;
        let arr1 = arr.slice();
        let res = [];
    
        for(let i = 0 ;i<n;i++){
            let ren = kernel_8(num);
            let itm = arr1[ren];
            res.push(itm);
            arr1.splice(ren,1);
            num = num -1;
        }
        return res;
    }
}
